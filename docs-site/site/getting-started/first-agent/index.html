async def convert_temperature(value: float, from_unit: str, to_unit: str) -> float:
    if from_unit == to_unit:
        return value

    if from_unit == "C" and to_unit == "F":
        return (value * 9/5) + 32

    else:
        raise ValueError(f"Unsupported units: {from_unit}, {to_unit}")
